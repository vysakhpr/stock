def home
  flash.keep
  if user_signed_in?
    if current_user.position=="HD"
      redirect_to "/hod"
    elsif current_user.position=="HM"
      redirect_to "/head"
    elsif current_user.position=="LC"
      redirect_to "/lab"
    else
      redirect_to "/office"
    end
      
  end
end
